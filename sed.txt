#!/bin/bash
# Apply fixes to all src/*.zig files
cd /Users/frankittermann/github/soteria-guard

# Fix ArrayList.init -> ArrayList.empty (the new unmanaged variant)
for f in src/git_log.zig src/scan.zig src/churn.zig src/complexity.zig src/ignore.zig src/main.zig src/thresholds.zig src/coupling.zig src/trend.zig; do
  python3 -c "
import re
with open('$f', 'r') as fh:
    content = fh.read()
# Replace .init(a) with .empty for ArrayList
content = content.replace('.init(a)', '.empty')
# Don't touch StringHashMap which still uses init
content = content.replace('StringHashMap(u32).empty', 'StringHashMap(u32).init(a)')
content = content.replace('StringHashMap(usize).empty', 'StringHashMap(usize).init(a)')
content = content.replace('StringHashMap(f64).empty', 'StringHashMap(f64).init(a)')
# Fix .init(allocator) for ignore.zig
content = content.replace('ArrayList([]const u8).empty', 'ArrayList([]const u8).init(allocator)')
# Fix back the ignore.zig one specifically
content = content.replace('.init(allocator).empty', '.init(allocator)')
content = content.replace('.init(std.heap.page_allocator)', '.empty')
content = content.replace('.init(allocator)', '.init(allocator)')  # no-op for StringHashMap only
with open('$f', 'w') as fh:
    fh.write(content)
"
done
echo "Done"
